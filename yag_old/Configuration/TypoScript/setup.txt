##########################################################
# Settings for yag extension
##########################################################
plugin.tx_yag {

    ##########################################################
    # Configuring the view
    ##########################################################
    view {
        templateRootPath = EXT:yag/Resources/Private/Templates/
        partialRootPath = EXT:yag/Resources/Private/Partials/
        layoutRootPath = EXT:yag/Resources/Private/Layouts/
    }
    
    
    ##########################################################
    # Configuring the persistence layer
    ##########################################################
    persistence {
        storagePid < plugin.tx_yag.settings.storagePid
    }
    

    #################################
    # Settings accessible in 
    # controllers via $this->settings
    #################################
    settings {
    
        # Comma-separated list of FE-Group UIDs that have admin privileges
        adminGroups = 1
    
        #############################
        # Album settings
        #############################
        album {
            # PID of page on which resides the plugin that generates media RSS
            rssPid = 5
            
            # How many items should be displayed on an album page
            itemsPerPage = 12

            #################################
            # Configuration for Album Actions
            #################################        
            actions {
                
                showMinimalisticAlbum {
                    display {
                        showDescription = 0
                        showPager = 0
                        showImageName = 0
                        showImageDescription = 0
                    }
                }
                
            }
            
        }
        
        # PID of page on which all records reside for this instance of yag gallery
        storagePid = 6
        
        #############################
        # JQuery settings
        #############################
        jquery {
            basePath = /fileadmin/jquery/jquery-1.3.2.js
            colorBoxPath = /fileadmin/jquery/colorbox/jquery.colorbox-min.js
        }
        
    }
    
}



##########################################################
# Settings for yag BE Module
##########################################################
module.tx_yag {
    settings < plugin.tx_yag.settings
    persistence < plugin.tx_yag.persistence
    view < plugin.tx_yag.view
    view {
        templateRootPath = EXT:yag/Resources/Private/Backend/Templates/
        partialRootPath = EXT:yag/Resources/Private/Backend/Partials/
        layoutRootPath = EXT:yag/Resources/Private/Backend/Layouts/
    }
}



##########################################################
# Settings for media RSS feed
##########################################################
xml = PAGE
xml {
  typeNum = 100
  10 = USER

            controller = Album
            actions = rss
        }
    }
    settings =< plugin.tx_yag.settings
    persistence =< plugin.tx_yag.persistence
    view =< plugin.tx_yag.view
  }
   
  config {
